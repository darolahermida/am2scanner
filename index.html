  1. Abrí script.google.com e iniciá sesión con tu cuenta Google.
  2. Clic en Nuevo proyecto.
  3. Borrá todo el código que hay y pegá el código de abajo.
  4. Clic en Guardar (ícono de disco). Ponele cualquier nombre al proyecto.
  5. Clic en Implementar → Nueva implementación.

  8. Pegá esa URL arriba y tocá Guardar, después Probar.
El script crea la planilla "AM2 Scanner - Registros" automáticamente en tu Google Drive la primera vez que guardes un registro. No necesitás crear nada vos.
Código para pegar en Apps Script: 
var PROPS = PropertiesService.getScriptProperties();

function getSheet() {
  var id = PROPS.getProperty('SHEET_ID');
  var ss;
  if (id) { try { ss = SpreadsheetApp.openById(id); } catch(e) { id = null; } }
  if (!id) {
    ss = SpreadsheetApp.create('AM2 Scanner - Registros');
    PROPS.setProperty('SHEET_ID', ss.getId());
    var s = ss.getActiveSheet();
    s.setName('Registros');
    s.appendRow(['Fecha','Hora','Cód.Artículo','Artículo','Cantidad','Unidad','Lote','Operario','Observaciones','Cód.Barras']);
    s.getRange(1,1,1,10).setFontWeight('bold').setBackground('#1a73e8').setFontColor('#ffffff');
    s.setFrozenRows(1);
  }
  return ss.getSheetByName('Registros') || ss.getActiveSheet();
}

function doPost(e) {
  try {
    var data = JSON.parse(e.postData.contents);
    var sheet = getSheet();
    var d = new Date(data.ts);
    sheet.appendRow([
      Utilities.formatDate(d, Session.getScriptTimeZone(), 'dd/MM/yyyy'),
      Utilities.formatDate(d, Session.getScriptTimeZone(), 'HH:mm:ss'),
      data.code, data.articulo, data.qty, data.unit,
      data.batch, data.worker, data.notes, data.barcode
    ]);
    var url = SpreadsheetApp.openById(PROPS.getProperty('SHEET_ID')).getUrl();
    return ContentService
      .createTextOutput(JSON.stringify({status:'ok', sheetUrl: url}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({status:'error', msg:err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function doGet() {
  var id = PROPS.getProperty('SHEET_ID');
  var url = '';
  if (id) { try { url = SpreadsheetApp.openById(id).getUrl(); } catch(e) {} }
  return ContentService
    .createTextOutput(JSON.stringify({status:'ok', sheetUrl: url}))
    .setMimeType(ContentService.MimeType.JSON);
}
